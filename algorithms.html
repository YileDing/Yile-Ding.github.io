📌 普通数列求和（7种方法）
# 方法1: for循环
def sum_for(n):
    total = 0
    for i in n:
        total += i
    return total

# 方法2: while循环
def sum_while(n):
    total = 0
    i = 0
    while i < len(n):
        total += n[i]
        i += 1
    return total

# 方法3: 递归
def sum_recursion(n):
    if n == []:
        return 0
    return n[0] + sum_recursion(n[1:])

# 方法4: 内置sum() [推荐]
def sum_builtin(n):
    return sum(n)

# 方法5: 列表推导式
def sum_comprehension(n):
    return sum([i for i in n])

# 方法6: 生成器表达式（节省内存）
def sum_generator(n):
    return sum((i for i in n))

# 方法7: 分治法
def sum_divide(n):
    if len(n) <= 1:
        return n[0] if n else 0
    mid = len(n) // 2
    return sum_divide(n[:mid]) + sum_divide(n[mid:])

📐 等差数列（O(1)公式法）
# 公式1：S = (首项 + 末项) × 项数 / 2
def sum_arithmetic(first, last, n):
    return (first + last) * n // 2

# 公式2：S = n × [2×首项 + (n-1)×公差] / 2
def sum_arithmetic_by_diff(first, diff, n):
    return n * (2 * first + (n - 1) * diff) // 2

# 示例：1+2+...+100 = 5050
print(sum_arithmetic(1, 100, 100))  # 输出: 5050
📐 等比数列（O(1)公式法）
def sum_geometric(first, q, n):
    """等比数列求和"""
    if q == 1:
        return first * n
    return first * (1 - q ** n) / (1 - q)

# 示例：1+2+4+8+16 = 31.0
print(sum_geometric(1, 2, 5))  # 输出: 31.0

💡 算法心得
1️⃣ 同一个问题可以有多种不同思路的解法
2️⃣ 数学公式往往比循环效率高得多（O(1) vs O(n)）
3️⃣ Python内置函数通常是最佳选择
4️⃣ 递归和分治虽然优雅，但实际使用中要权衡效率
