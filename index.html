        self.gpa = 50          # 学习成绩

        self.mental = 50       # 心理健康
        self.social = 10       # 社会实践
        self.money = 500       # 金钱
        self.month = 1         # 当前月份
        self.total_months = 24 # 总时长2学年

    def get_progress_bar(self, value, label, bar_char='#'):
        """绘制进度条"""
        bar_length = 20
        percent = value / 100
        filled_length = int(bar_length * percent)
        bar = bar_char * filled_length + '-' * (bar_length - filled_length)
        return f"{label}: [{bar}] {value}"

数列求和：一题多解

算法思想 时间复杂度分析

探索同一问题的多种解法，涵盖迭代、递归、分治、数学公式等不同算法思想。 对比各方法的时间复杂度与适用场景。

#for循环 #递归 #分治法 #数学公式(O(1)) #生成器

📌 普通数列求和（7种方法）

# 方法1: for循环
def sum_for(n):
    total = 0
    for i in n:
        total += i
    return total

# 方法2: while循环
def sum_while(n):
    total = 0
    i = 0
    while i < len(n):
        total += n[i]
        i += 1
    return total

# 方法3: 递归
def sum_recursion(n):
    if n == []:
        return 0
    return n[0] + sum_recursion(n[1:])

# 方法4: 内置sum() [推荐]
def sum_builtin(n):
    return sum(n)

# 方法5: 列表推导式
def sum_comprehension(n):
    return sum([i for i in n])

# 方法6: 生成器表达式（节省内存）
def sum_generator(n):
    return sum((i for i in n))

# 方法7: 分治法
def sum_divide(n):
    if len(n) <= 1:
        return n[0] if n else 0
    mid = len(n) // 2
    return sum_divide(n[:mid]) + sum_divide(n[mid:])

📊 方法对比总结

方法时间复杂度优点适用场景
for/while循环O(n)直观易懂通用场景
递归O(n)代码简洁函数式编程
内置sum()O(n)C语言实现，最快🌟 首选方法
生成器O(n)内存高效大数据集
分治法O(n log n)并行计算思想学习分治策略

📐 特殊数列：等差数列（O(1)公式法）

# 公式1：S = (首项 + 末项) × 项数 / 2
def sum_arithmetic(first, last, n):
    return (first + last) * n // 2

# 公式2：S = n × [2×首项 + (n-1)×公差] / 2
def sum_arithmetic_by_diff(first, diff, n):
    return n * (2 * first + (n - 1) * diff) // 2

# 示例：1+2+...+100 = 5050

📐 特殊数列：等比数列（O(1)公式法）

def sum_geometric(first, q, n):
    """等比数列求和"""
    if q == 1:
        return first * n
    return first * (1 - q ** n) / (1 - q)

# 示例：1+2+4+8+16 = 31
💡 学习心得
1️⃣ 同一个问题可以有多种不同思路的解法
2️⃣ 数学公式往往比循环效率高得多（O(1) vs O(n)）
3️⃣ Python内置函数通常是最佳选择
4️⃣ 递归和分治虽然优雅，但实际使用中要权衡效率

✅ 运行结果验证

>>> sum_for([1,2,3,4,5])
15
>>> sum_arithmetic(1, 100, 100)
5050
>>> sum_geometric(1, 2, 5)
31.0
